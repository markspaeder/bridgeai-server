<script>
const BAI_SERVER = 'YOUR_BRIDGEAI_RAILWAY_URL';
const BAI = { history: [], busy: false };
const BAI_WELCOME = "Hi! I'm the Bridge AI assistant. We build custom AI agents for trade businesses — contractors, HVAC, pool companies, real estate, and more. Every agent captures leads 24/7 and notifies you instantly. What can I help you with?";
const BAI_QUICK = ['See pricing', 'How does it work?', 'Get a demo', 'Call us: 754-444-7550'];

function baiAddMsg(role, text) {
  const msgs = document.getElementById('bai-messages');
  const div = document.createElement('div');
  div.className = 'bai-msg ' + role;
  div.innerHTML = '<div class="bai-bubble">' + text.replace(/\n/g,'<br>') + '</div>';
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
}
function baiShowTyping() {
  const msgs = document.getElementById('bai-messages');
  const div = document.createElement('div');
  div.className = 'bai-msg bot'; div.id = 'bai-typing';
  div.innerHTML = '<div class="bai-bubble bai-typing"><span></span><span></span><span></span></div>';
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
}
function baiHideTyping() { const t = document.getElementById('bai-typing'); if(t) t.remove(); }
function baiShowQuick(replies) {
  const q = document.getElementById('bai-quick'); q.innerHTML = '';
  replies.forEach(r => {
    const btn = document.createElement('button');
    btn.className = 'bai-qbtn'; btn.textContent = r;
    btn.onclick = () => { q.innerHTML = ''; baiProcess(r); };
    q.appendChild(btn);
  });
}
async function baiProcess(text) {
  if(!text.trim() || BAI.busy) return;
  BAI.busy = true;
  document.getElementById('bai-quick').innerHTML = '';
  baiAddMsg('user', text);
  BAI.history.push({role:'user', content:text});
  baiShowTyping();
  if(text.toLowerCase().includes('call')) {
    baiHideTyping();
    const r = "You can reach Mark directly at <strong>754-444-7550</strong> or email <strong>leads@mybridgeai.com</strong>. Happy to answer any other questions!";
    baiAddMsg('bot', r);
    BAI.history.push({role:'assistant', content:r});
    BAI.busy = false; return;
  }
  try {
    const res = await fetch(BAI_SERVER + '/chat', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({messages: BAI.history, client: 'bridgeai'})
    });
    const data = await res.json();
    baiHideTyping();
    const reply = data.text || "Having trouble connecting. Call us at 754-444-7550.";
    baiAddMsg('bot', reply);
    BAI.history.push({role:'assistant', content:reply});
  } catch(e) {
    baiHideTyping();
    baiAddMsg('bot', "Having trouble connecting. Call us at 754-444-7550.");
  }
  BAI.busy = false;
}
function baiSend() {
  const input = document.getElementById('bai-input');
  const text = input.value.trim();
  input.value = ''; input.style.height = 'auto';
  if(text) baiProcess(text);
}
function baiMinimize() {
  document.getElementById('bai-wrap').style.display = 'none';
  const r = document.getElementById('bai-restore');
  r.style.display = 'flex';
}
function baiExpand() {
  document.getElementById('bai-wrap').style.display = 'block';
  document.getElementById('bai-restore').style.display = 'none';
}
document.getElementById('bai-input').addEventListener('keydown', function(e) {
  if(e.key==='Enter' && !e.shiftKey) { e.preventDefault(); baiSend(); }
});
document.getElementById('bai-input').addEventListener('input', function() {
  this.style.height='auto'; this.style.height=Math.min(this.scrollHeight,90)+'px';
});
setTimeout(() => {
  baiAddMsg('bot', BAI_WELCOME);
  BAI.history.push({role:'assistant', content:BAI_WELCOME});
  baiShowQuick(BAI_QUICK);
}, 600);
</script>
